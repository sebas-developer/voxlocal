This example demonstrates real-time TTS with streaming playback.

## Simple Real-time Playback

```python
from voxlocal import VoxLocal

vl = VoxLocal("en")
vl.setup()

# Play text in real-time
vl.play("Hello! This is real-time text to speech.")
```

## Streaming with Progress

```python
from voxlocal import VoxLocal
from voxlocal.playback import PlaybackEvent

vl = VoxLocal("en")
vl.setup()

def on_event(event: PlaybackEvent):
    print(f"[{event.elapsed_seconds:.1f}s] Chunk {event.sequence}")
    if event.final:
        print("Playback complete!")

vl.play(
    "This demonstrates real-time streaming with progress callbacks.",
    on_event=on_event,
)
```

## Manual Streaming

For full control over streaming:

```python
from voxlocal import VoxLocal
import sounddevice as sd

vl = VoxLocal("en")
vl.setup()

# Get streaming chunks
stream = vl.stream("Manual streaming gives you full control.")

# Play through sounddevice
with sd.OutputStream(
    samplerate=22050,
    channels=1,
    dtype="float32",
) as audio_stream:
    for chunk in stream:
        print(f"Playing chunk {chunk.sequence}")
        audio_stream.write(chunk.numpy)
```

## Producer-Consumer Pattern

For advanced use cases:

```python
from voxlocal import VoxLocal
from queue import Queue
import threading
import sounddevice as sd

vl = VoxLocal("en")
vl.setup()

audio_queue = Queue()

def producer():
    """Generate audio chunks."""
    for chunk in vl.stream("Producer-consumer pattern for streaming."):
        audio_queue.put(chunk.numpy)
    audio_queue.put(None)  # Signal completion

def consumer():
    """Play audio chunks."""
    with sd.OutputStream(
        samplerate=22050,
        channels=1,
        dtype="float32",
    ) as stream:
        while True:
            data = audio_queue.get()
            if data is None:
                break
            stream.write(data)

# Start both threads
producer_thread = threading.Thread(target=producer, daemon=True)
consumer_thread = threading.Thread(target=consumer, daemon=True)

producer_thread.start()
consumer_thread.start()

producer_thread.join()
consumer_thread.join()

print("Done!")
```

## Multi-language Real-time

```python
from voxlocal import VoxLocal

# Create instances for different languages
languages = {
    "en": VoxLocal("en"),
    "es": VoxLocal("es"),
    "fr": VoxLocal("fr"),
}

# Setup all
for vl in languages.values():
    vl.setup()

# Play in different languages
texts = {
    "en": "Hello in English!",
    "es": "Hola en español!",
    "fr": "Bonjour en français!",
}

for lang, text in texts.items():
    print(f"Speaking {lang}...")
    languages[lang].play(text)
```

## Error Handling

```python
from voxlocal import VoxLocal
from voxlocal._errors import DependencyMissingError, SynthesisError

vl = VoxLocal("en")
vl.setup()

try:
    vl.play("Real-time playback!")
except DependencyMissingError:
    print("Install sounddevice: pip install 'voxlocal[playback]'")
except SynthesisError as e:
    print(f"Synthesis failed: {e}")
```

## Complete Example

```python
#!/usr/bin/env python3
"""Real-time TTS example with VoxLocal."""

from voxlocal import VoxLocal
from voxlocal.playback import PlaybackEvent

def main():
    # Initialize
    print("Initializing VoxLocal...")
    vl = VoxLocal("en")
    vl.setup()

    # Text to speak
    text = """
    This is a complete example of real-time text to speech
    using VoxLocal. The audio streams progressively as it
    is generated, providing low-latency playback.
    """

    # Progress callback
    def on_event(event: PlaybackEvent):
        if event.kind == "playing":
            print(f"Playing chunk {event.sequence} ({event.audio_seconds:.2f}s)")

    # Play
    print("Starting playback...")
    vl.play(text.strip(), on_event=on_event)
    print("Playback complete!")

if __name__ == "__main__":
    main()
```
